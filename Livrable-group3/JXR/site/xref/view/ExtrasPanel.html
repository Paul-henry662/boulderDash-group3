8   import java.awt.Rectangle;
9   import java.util.Observable;
10  import java.util.Observer;

12  import javax.swing.JPanel;
13  
14  class ExtrasPanel extends JPanel implements Observer{
15  	/** The view frame. */
16  	private ViewFrame					viewFrame;
17  	
18  	/** The Constant serialVersionUID. */
19  	private static final long			serialVersionUID	= -998294702363713521L;
20  	
21  	/** The width of the panel */
22  	private int width;
23  	
24  	/** The height of the panel */
25  	private int height;
26  
27  	/**
28  	 * Instantiates a new view panel.
29  	 *
30  	 * @param viewFrame
31  	 *          the view frame
32  	 */
33  	public ExtrasPanel(final ViewFrame viewFrame) {
34  		this.setViewFrame(viewFrame);
35  		this.width = this.getViewFrame().getModel().getMap().getWidth()*View.SQUARE_SIZE;
36  		this.height = (this.getViewFrame().getModel().getMap().getHeight()*View.SQUARE_SIZE);
37  		viewFrame.getModel().getObservable().addObserver(this);
38  	}
39  
40  	/**
41  	 * Gets the view frame.
42  	 *
43  	 * @return the view frame
44  	 */
45  	private ViewFrame getViewFrame() {
46  		return this.viewFrame;
47  	}
48  
49  	/**
50  	 * Sets the view frame.
51  	 *
52  	 * @param viewFrame
53  	 *          the new view frame
54  	 */
55  	private void setViewFrame(final ViewFrame viewFrame) {
56  		this.viewFrame = viewFrame;
57  	}
58  	
59  	
60  	private void paintScore(Graphics graphics) {
61  		graphics.drawString("Score: "+this.getViewFrame().getModel().getScore(), 20, 20);
62  	}
63  	
64  	/*
65  	 * (non-Javadoc)
66  	 *
67  	 * @see java.util.Observer#update(java.util.Observable, java.lang.Object)
68  	 */
69  	public void update(final Observable arg0, final Object arg1) {
70  		this.repaint();
71  	}
72  	
73  	private void paintTime(Graphics graphics) {
74  		graphics.drawString("Time: "+this.getViewFrame().getModel().getTimer(), 100, 20);
75  	}
76  	
77  	/*
78  	 * (non-Javadoc)
79  	 *
80  	 * @see javax.swing.JComponent#paintComponent(java.awt.Graphics)
81  	 */
82  	@Override
83  	protected void paintComponent(final Graphics graphics) {	
84  		Graphics2D graphics2d = (Graphics2D) graphics;
85  		graphics2d.setPaint(Color.white);
86  		graphics2d.setFont(new Font("Arial", 20,20));
87  		this.paintScore(graphics2d);
88  		this.paintTime(graphics2d);
89  	}
90  }
